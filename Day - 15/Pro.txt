03) Print numbers from 1 to 10 and 1 to 5 should be print by child process and from 6 to 10 should be parent process calculate summation of those numbers

#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>

int main() {

    int f = fork();
    int sum1 = 0, 
	int sum2 = 0, 
	int sum = 0;
    
    if (f == 0) { 
        for (int i = 1; i <= 5; i++) {
            printf("%d\n", i);
            sum1 += i;
        }
        return sum1; 
    } 
    else { 
        int status;
        wait(&status);
        
        sum1 = WEXITSTATUS(status); 
        
        for (int j = 6; j <= 10; j++) {
            printf("%d\n", j);
            sum2 += j;
        }
        
        sum = sum1 + sum2;
        printf("Summation: %d\n", sum);
    }
    return 0;
}
